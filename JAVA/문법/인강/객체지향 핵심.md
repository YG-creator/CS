   ```java

       public Child() {
           // super(); 자동으로 부모 기본생성자로 생성됨
           // 추가 구현
       }
       
       public Child(int a) {
           // super(); 자동으로 부모 기본생성자로 생성되므로 수정이 필요
           super(int a);
           // 추가 구현
       }
   }
   ```
